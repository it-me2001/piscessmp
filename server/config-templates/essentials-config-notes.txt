# EssentialsX — homes & teleport (plugins/Essentials/config.yml)
#
# After first start, tune these if needed:
#
#   spawn-on-join: true
#   spawns.default.world: spawn   (hub — separate from survival overworld)
#   teleport-cooldown: 5
#   teleport-delay: 3
#
# Home limits are controlled by LuckPerms (see luckperms-homes-rtp.txt):
#   essentials.sethome.multiple.3  → 3 homes
#   essentials.sethome.multiple.5  → 5 homes
#   essentials.sethome.multiple.10 → 10 homes
#
# Player commands:
#   /sethome [name]   — save a home (default name: home)
#   /home [name]      — teleport to a saved home
#   /delhome [name]   — delete a home
#   /homes            — list your homes
#   /spawn            — teleport to hub (from anywhere)
#   /tpa <player>     — request to teleport to someone
#   /tpaccept         — accept a TPA request
#   /tpdeny           — deny a TPA request
